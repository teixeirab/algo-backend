select
concat(series_number, '-', t0.account_number) 'Account',
security_id 'Symbol',
cusip 'ID',
asset_classification 'Asset Type',
'USD' as 'Currency',
quantity 'Quantity',
price 'Market Price',
market_value 'Market Value',
description 'Description'
from pershing_positions t0,
series_product_information t1
where period = '{param0}'
and t0.account_number = t1.account_number
