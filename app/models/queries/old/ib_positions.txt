select
concat(series_number, '-', account_id) 'Account',
symbol 'Symbol',
bb_global_id 'ID',
asset_type 'Asset Type',
t0.currency 'Currency',
quantity 'Quantity',
market_price 'Market Price',
market_value 'Market Value'
from ib_positions t0,
series_product_information t1
where report_date = '{param0}'
and t0.account_id = t1.account_number
