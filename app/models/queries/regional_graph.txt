select
'Nominal Issued',
(select sum(nominal_amount) from trades t, series_product_information s where t.series_number = s.series_number and trade_type = 'Issuance' and region = '{param0}' and settlement_date >= DATE_SUB(now(), INTERVAL 1 MONTH)) Last_Month,
(select sum(nominal_amount) from trades t, series_product_information s where t.series_number = s.series_number and trade_type = 'Issuance' and region = '{param0}' and settlement_date >= DATE_FORMAT(NOW() ,'%Y-01-01') ) Year_to_date,
(select sum(nominal_amount) from trades t, series_product_information s where t.series_number = s.series_number and trade_type = 'Issuance' and region = '{param0}' and settlement_date >= DATE_SUB(now(), INTERVAL 12 MONTH)) Trailing_Twelve_Months,
(select sum(nominal_amount) from trades t, series_product_information s where t.series_number = s.series_number and trade_type = 'Issuance' and region = '{param0}') Total
Union
# Active Notes

(select sum(nominal_outstanding) from trades t, series_product_information s where t.series_number = s.series_number and region = '{param0}' and settlement_date >= DATE_FORMAT(NOW() ,'%Y-01-01') ) Year_to_date,
(select sum(nominal_outstanding) from trades t, series_product_information s where t.series_number = s.series_number and region = '{param0}' and settlement_date >= DATE_SUB(now(), INTERVAL 12 MONTH) ) Trailing_Twelve_Months,
(select sum(nominal_outstanding) from trades t, series_product_information s where t.series_number = s.series_number and region = '{param0}')Total

# Active Funded

(select sum(cash_proceeds) from trades t, series_product_information s where t.series_number = s.series_number and region = '{param0}') Total
Union
# Numbe of Issuances
select
'Number of Issuances',
(select count(1) from series_product_information where region = '{param0}' and issue_date >= DATE_SUB(now(), INTERVAL 1 MONTH) ) Last_Month,
(select count(1) from series_product_information where region = '{param0}' and issue_date >= DATE_FORMAT(NOW() ,'%Y-01-01') ) Year_to_date,
(select count(1) from series_product_information where region = '{param0}' and issue_date >= DATE_SUB(now(), INTERVAL 12 MONTH) ) Trailing_Twelve_Months,
(select count(1) from series_product_information where region = '{param0}') Total
